function fibonacci ( n : integer ) : integer {
    if ( n == 0 ) return 1;
    if ( n == 1 ) return 1;
    return fibonacci(n-1) + fibonacci(n-2);
}

procedure main()
{
   print_int(fibonacci(8));
}
